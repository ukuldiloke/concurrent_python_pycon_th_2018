Multithread

Pro

  • Light weight
  • Good for I/O related tasks
  • Easy data sharing

Con

  • Addressable memory is limited to the max memory of a single process
  • Not suitable for computing heavy tasks
  • Has limitation from GIL

              from concurrent.futures import ThreadPoolExecutor
              import requests
              
              URLS = [
                'https://www.kaidee.com',
                'https://www.python.org',
                'https://www.google.com',
              ]
              
              def fetch(url):
                print('Fetching {}'.format(url))
                resp = requests.get(url)
                return resp.text
             
              if __name__ == '__main__':
                with ThreadPoolExecutor(max_workers=2) as executor:
                  for url, data in zip(URLS, executor.map(fetch, URLS)):
                    print('{} page is {} bytes'.format(url, len(data)))

              from concurrent.futures import ProcessPoolExecutor
              import math
              
              PRIMES = [
                112272535095293,
                112582705942171,
                112272535095293,
                115280095190773,
                115797848077099,
                1099726899285419,
              ]
              
              def is_prime(n):
                if n % 2 == 0:
                  return False
              
                sqrt_n = int(math.floor(math.sqrt(n)))
                for i in range(3, sqrt_n + 1, 2):
                  if n % i == 0:
                    return False
                return True

              if __name__ == '__main__':
                with ProcessPoolExecutor(max_workers=4) as executor:
                  for number, prime in zip(PRIMES, executor.map(is_prime, PRIMES)):
                    print('{} is prime: {}'.format(number, prime))

Eventloop

Code example


                import aiohttp
                import asyncio
                
                URLS = [
                  'https://www.kaidee.com',
                  'https://www.python.org',
                  'https://www.google.com',
                ]
                
                async def fetch(session, url):
                  print('Fetching {}'.format(url))
                  async with session.get(url) as response:
                    return await response.text()
                
                async def main():
                  async with aiohttp.ClientSession() as session:
                    coros = [fetch(session, url) for url in URLS]
                    results = await asyncio.gather(*coros)
                    for url, data in zip(URLS, results):
                      print('{} page is {} bytes'.format(url, len(data)))
               
                if __name__ == '__main__':
                  loop = asyncio.get_event_loop()
                  loop.run_until_complete(main())
